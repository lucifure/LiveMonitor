name: Build YouTube Live Monitor APK

on:
  push:
    branches: [main, master]
    paths:
      - 'kivy-app/**'
      - '.github/workflows/build-apk.yml'
  workflow_dispatch:        # lets you trigger a build manually from the Actions tab

jobs:
  build:
    name: Build debug APK
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Set up Python 3.11
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Set up JDK 17
        uses: actions/setup-java@v4
        with:
          distribution: 'temurin'
          java-version: '17'

      - name: Cache buildozer global dir
        uses: actions/cache@v4
        with:
          path: ~/.buildozer
          key: buildozer-global-${{ hashFiles('kivy-app/buildozer.spec') }}
          restore-keys: buildozer-global-

      - name: Cache buildozer local dir
        uses: actions/cache@v4
        with:
          path: kivy-app/.buildozer
          key: buildozer-local-${{ hashFiles('kivy-app/buildozer.spec') }}
          restore-keys: buildozer-local-

      - name: Build APK with buildozer
        uses: ArtemSBulgakov/buildozer-action@v1
        id: buildozer
        with:
          command: buildozer android debug
          workdir: kivy-app
          buildozer_version: stable

      - name: Upload APK artifact
        uses: actions/upload-artifact@v4
        with:
          name: youtubelivemonitor-debug-apk
          path: ${{ steps.buildozer.outputs.filename }}
          if-no-files-found: error
